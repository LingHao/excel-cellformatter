11  /**
12   * 数字の表現を大字に変換する。
13   * <p>数字が連続しないと、桁数を考慮した変換はできない。
14   * <p>途中に区切り文字などがあると、そこで途切れる。
15   *
16   * @version 0.5
17   * @author T.TSUCHIE
18   *
19   */
20  public class DaijiCallback extends JapaneseCallback<Object> {
21      
22      /**
23       * インスタンスを取得する
24       * @return
25       */
26      public static DaijiCallback create() {
27          return new DaijiCallback();
28      }
29      
30      /**
31       * 整数部分の切り出し用正規表現
32       */
33      private static final Pattern PATTERN_NUM = Pattern.compile("([\\D]*)([\\d]+)([\\.]{0,1}[.\\s\\w]*)");
34      
35      @Override
36      public String call(final Object data, final String value, final Locale locale) {
37          
38          final Matcher matcher = PATTERN_NUM.matcher(value);
39          if(!matcher.matches()) {
40              // 単純に数値の変換
41              return replaceSimple(value);
42          }
43          
44          final String before = matcher.group(1);
45          final String num = matcher.group(2);
46          final String after = matcher.group(3);
47          
48          return replaceSimple(before) + replaceDisits(num) + replaceSimple(after);
49      }
50      
51      /**
52       * 数字のマップ
53       */
54      private static final String[][] NUM_MAP = {
55          {"0", "〇"},
56          {"1", "壱"},
57          {"2", "弐"},
58          {"3", "参"},
59          {"4", "四"},
60          {"5", "伍"},
61          {"6", "六"},
62          {"7", "七"},
63          {"8", "八"},
64          {"9", "九"},
65      };
66      
67      /**
68       * 数字を単純に変換する。
69       * @param value
70       * @return
71       */
72      private String replaceSimple(final String value) {
73          String str = value;
74          for(String[] item : NUM_MAP) {
75              str = str.replaceAll(item[0], item[1]);
76          }
77          
78          return str;
79      }
80      
81      /**
82       * 10^4ごとの桁数のマップ
83       * <p><a href="http://www2s.biglobe.ne.jp/~hotori/kazu.html">桁数の名前</a>
84       */
85      private static final String[] DIGITS_10_MAP = {
86          "",      // 10^0
87          "萬",    // 10^4
88          "億",    // 10^8
89          "兆",    // 10^12
90          "京",    // 10^16
91      };
92      
93      /**
94       * 数字を桁数に合わせて変換する。
95       * @param value
96       * @return
97       */
98      private String replaceDisits(final String value) {
99          
100         if(value.equals("0")) {
101             return NUM_MAP[0][1];
102         }
103         
104         // 4桁ごとに、分割する。
105         final int length = value.length();
106         final List<String> split4 = new ArrayList<>();
107         for(int i=0; i < length; i=i+4) {
108             
109             // 下の桁から切り出す
110             int end = length -i;
111             int start;
112             if(i + 4 < length) {
113                 start = end - 4;
114             } else {
115                 start = 0;
116             }
117             String item = value.substring(start, end);
118             split4.add(item);
119         }
120         
121         // 4桁ごとに変換を行う。
122         final List<String> digits = new ArrayList<>();
123         for(int i=0; i < split4.size(); i++) {
124             String item = split4.get(i);
125             item = replace4Digits(item) + DIGITS_10_MAP[i];
126             
127             digits.add(item);
128         }
129         
130         /*
131          * 文字列に直す。
132          * ・桁数が逆順になっているので、戻し結合する。
133          */
134         Collections.reverse(digits);
135         StringBuilder sb = new StringBuilder();
136         for(String item : digits) {
137             sb.append(item);
138         }
139         
140         return sb.toString();        
141     }
142     
143     /**
144      * 4桁の桁数のマップ
145      */
146     private static final String[] DIGITS_4_MAP = {
147         "",
148         "拾",
149         "百",
150         "阡",
151     };
152     
153     /**
154      * 4桁以下の数字を、漢数字に変換する。
155      * <p>大字の場合は、千の位の１は省略しない。
156      * @param value
157      * @return
158      */
159     private String replace4Digits(final String value) {
160         
161         // 桁ごとに変換を行う。
162         final int length = value.length();
163         final List<String> digits = new ArrayList<>();
164         
165         for(int i=0; i < length; i++) {
166             // 下の桁から処理する
167             final char c = value.charAt(length-i-1);
168             if(c == '0') {
169                 continue;
170                 
171             }
172             
173             String item;
174             if(c == '1' && (0 < i && i < 2)) {
175                 // 10の位以上で、かつ1の場合、数字部分は省略し、桁数のみにする。
176 //                item = DIGITS_4_MAP[i];
177                 //Excelの場合は、12の場合も、「壱拾弐」と1を省略さない。
178                 item = replaceSimple(String.valueOf(c)) + DIGITS_4_MAP[i];
179             } else {
180                 item = replaceSimple(String.valueOf(c)) + DIGITS_4_MAP[i];
181             }
182             
183             digits.add(item);
184             
185         }
186         
187         /*
188          * 文字列に直す。
189          * ・桁数が逆順になっているので、戻し結合する。
190          */
191         Collections.reverse(digits);
192         StringBuilder sb = new StringBuilder();
193         for(String item : digits) {
194             sb.append(item);
195         }
196         
197         return sb.toString();
198         
199     }
