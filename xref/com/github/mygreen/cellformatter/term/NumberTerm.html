53      /**
54       * フォーマットの書式"General"を表現する項
55       *
56       */
57      public static class GeneralTerm extends NumberTerm {
58          
59          private final ThreadLocal<DecimalFormat> formatter = new ThreadLocal<DecimalFormat>() {
60              
61              @Override
62              protected DecimalFormat initialValue() {
63                  final DecimalFormat format = new DecimalFormat("0.##########");
64                  format.setRoundingMode(RoundingMode.HALF_UP);
65                  return format;
66              }
67              
68          };
69          
70          @Override
71          public String format(final FormattedNumber number, final MSLocale formatLocale, final Locale runtimeLocale) {
72              return formatter.get().format(Math.abs(number.getValue()));
73          }
74          
75      }

77      /**
78       * 数値のフォーマット部分の項を表す抽象クラス。
79       *
80       */
81      public static abstract class FormattedTerm extends NumberTerm {
82          
83          /** 桁のインデックス */
84          protected int index;
85          
86          /** 書式の部分 */
87          protected NumberPartType partType;
88          
89          /** 書式の部分の最後かどうか */
90          protected boolean lastPart;
91          
92          /** 桁の区切り文字を出力するかどうか */
93          protected boolean outSepearator;
94          
95          public FormattedTerm index(final int index) {
96              this.index = index;
97              return this;
98          }
99          
100         public FormattedTerm partType(final NumberPartType partType) {
101             this.partType = partType;
102             return this;
103         }
104         
105         public FormattedTerm lastPart(final boolean lastPart) {
106             this.lastPart = lastPart;
107             return this;
108         }
109         
110         /**
111          * 数値の部分に対する桁の値を取得する。
112          * @param number
113          * @return
114          */
115         protected String getNumber(final FormattedNumber number) {
116             
117             switch(partType) {
118                 case Integer:
119                     if(isLastPart()) {
120                         return number.asDecimal().getIntegerPartAfter(getIndex());
121                     } else {
122                         return number.asDecimal().getIntegerPart(getIndex());
123                     }
124                     
125                 case Decimal:
126                     return number.asDecimal().getDecimalPart(getIndex());
127                     
128                 case Exponent:
129                     if(isLastPart()) {
130                         return number.asExponent().getExponentPartAfter(getIndex());
131                     } else {
132                         return number.asExponent().getExponentPart(getIndex());
133                     }
134                 case Denominator:
135                     if(isLastPart()) {
136                         return number.asFraction().getDenominatorPartAfter(getIndex());
137                     } else {
138                         return number.asFraction().getDenominatorPart(getIndex());
139                     }
140                     
141                 case Numerator:
142                     
143                     if(isLastPart()) {
144                         return number.asFraction().getNumeratorPartAfter(getIndex());
145                     } else {
146                         return number.asFraction().getNumeratorPart(getIndex());
147                     }
148                     
149                 case WholeNumber:
150                     if(isLastPart()) {
151                         return number.asFraction().getWholeNumberPartAfter(getIndex());
152                     } else {
153                         return number.asFraction().getWholeNumberPart(getIndex());
154                     }
155                     
156                 default:
157                     return "";
158             }
159             
160             
161         }
162         
163         public int getIndex() {
164             return index;
165         }
166         
167         public void setIndex(int index) {
168             this.index = index;
169         }
170         
171         public NumberPartType getPartType() {
172             return partType;
173         }
174         
175         public void setPart(NumberPartType partType) {
176             this.partType = partType;
177         }
178         
179         public boolean isLastPart() {
180             return lastPart;
181         }
182         
183         public void setLastPart(boolean lastPart) {
184             this.lastPart = lastPart;
185         }
186         
187         public boolean isOutSepearator() {
188             return outSepearator;
189         }
190         
191         public void setOutSepearator(boolean outSepearator) {
192             this.outSepearator = outSepearator;
193         }
194         
195     }

197     /**
198      * フォーマットの書式"0"の記号。
199      * ・出力する項がない場合は、0を出力する。
200      */
201     public static class ZeroTerm extends FormattedTerm {
202         
203         private static final String ZERO = "0";
204         
205         @Override
206         public String format(final FormattedNumber number, final MSLocale formatLocale, final Locale runtimeLocale) {
207             
208             String num = getNumber(number);
209             if(num.isEmpty()) {
210                 return ZERO;
211             }
212             
213             return num;
214         }

254     /**
255      * フォーマットの書式の指数"E"を表現する項。
256      * ・指数部の符号も出力する。
257      *
258      */
259     public static class ExponentTerm extends NumberTerm {
260         
261         /**
262          * 符号
263          * ・ただし、符号がない場合がある。
264          * ・出力するときには、設定された符号は無視する。
265          */
266         private final Token token;
267         
268         /**
269          * 指数の記号。
270          * ・パターンによって、大文字、小文字がある。
271          */
272         private final String exponentSymbol;
273         
274         public ExponentTerm(final Token token) {
275             this.token = token;
276             
277             final String vale = token.getValue();
278             if(vale.startsWith("E")) {
279                 this.exponentSymbol = "E";
280             } else {
281                 this.exponentSymbol = "e";
282             }
283         }
284         
285         @Override
286         public String format(final FormattedNumber number, final MSLocale formatLocale, final Locale runtimeLocale) {
287             
288             if(number.asExponent().isExponentPositive()) {
289                 if(Utils.startsWithIgnoreCase(getToken().getValue(), "E-")) {
290                     // 指数がマイナスの場合は、正の時に符号は付与しない。
291                     return exponentSymbol;
292                 } else {
293                     return exponentSymbol + "+";
294                 }
295             } else {
296                 return exponentSymbol + "-";
297             }
298         }
299         
300         public Token getToken() {
301             return token;
302         }
303         
304         public String getExponentSymbol() {
305             return exponentSymbol;
306         }
307         
308     }
